# 08 - 环境上下文采集

> **目的**：采集 agent 分析所需的"环境事实"类上下文，区别于 01-07 的时序数据，这类信息更新频率低（一次配置、长期使用），但所有 skill 都依赖。
> **最终产物**：直接更新 `memory/` 目录下 4 个 JSON 文件
> - `memory/sql_config.json`（自动采集）
> - `memory/d365_custom.json`（半自动 - 采集脚本给建议，人工确认）
> - `memory/business_context.json`（手动 - 业务/运维填写）
> - `memory/thresholds.json`（手动 - 按客户调整默认模板）

## 一、sql_config.json — 完全自动采集

### 1.1 SQL 采集脚本

保存为 `C:\DBA\scripts\Collect-SqlConfig.sql`，在每个 SQL 实例上执行。脚本输出一组 CSV（由 PowerShell 包装执行后合并为 JSON）。

```sql
/* ============ Collect-SqlConfig.sql ============ */
SET NOCOUNT ON;

/* 1) 实例版本 */
SELECT
    @@SERVERNAME                              AS ServerName,
    SERVERPROPERTY('InstanceName')            AS InstanceName,
    SERVERPROPERTY('ProductVersion')          AS ProductVersion,
    SERVERPROPERTY('ProductLevel')            AS ProductLevel,
    SERVERPROPERTY('Edition')                 AS Edition,
    SERVERPROPERTY('IsHadrEnabled')           AS AlwaysOnEnabled;
GO

/* 2) sp_configure 关键项 */
SELECT name, value_in_use
FROM sys.configurations
WHERE name IN (
    'max server memory (MB)',
    'min server memory (MB)',
    'max degree of parallelism',
    'cost threshold for parallelism',
    'optimize for ad hoc workloads',
    'max worker threads',
    'backup compression default',
    'fill factor (%)',
    'remote query timeout (s)'
);
GO

/* 3) tempdb 文件布局 */
SELECT
    DB_NAME(mf.database_id) AS DatabaseName,
    mf.name                 AS LogicalName,
    mf.type_desc            AS FileType,
    mf.size * 8 / 1024      AS SizeMB,
    CASE WHEN mf.max_size = -1 THEN -1 ELSE mf.max_size * 8 / 1024 END AS MaxSizeMB,
    CASE WHEN mf.is_percent_growth = 1 THEN mf.growth ELSE mf.growth * 8 / 1024 END AS GrowthMB,
    mf.physical_name        AS PhysicalPath
FROM sys.master_files mf
WHERE mf.database_id = 2;
GO

/* 4) 数据库配置（重点 MSCRM） */
SELECT
    d.name,
    d.is_read_committed_snapshot_on,
    d.snapshot_isolation_state_desc,
    d.recovery_model_desc,
    d.compatibility_level,
    d.is_auto_shrink_on,
    d.is_auto_close_on,
    d.is_auto_create_stats_on,
    d.is_auto_update_stats_on,
    d.is_auto_update_stats_async_on,
    d.page_verify_option_desc,
    d.is_trustworthy_on,
    d.is_broker_enabled
FROM sys.databases d
WHERE d.name LIKE 'MSCRM%' OR d.database_id <= 4;
GO

/* 5) 数据库大小 */
SELECT
    DB_NAME(mf.database_id) AS DatabaseName,
    SUM(CASE WHEN mf.type_desc='ROWS' THEN mf.size ELSE 0 END) * 8 / 1024 AS DataSizeMB,
    SUM(CASE WHEN mf.type_desc='LOG'  THEN mf.size ELSE 0 END) * 8 / 1024 AS LogSizeMB
FROM sys.master_files mf
WHERE DB_NAME(mf.database_id) LIKE 'MSCRM%'
GROUP BY DB_NAME(mf.database_id);
GO

/* 6) 全局 Trace Flag */
DBCC TRACESTATUS(-1) WITH NO_INFOMSGS;
GO

/* 7) AlwaysOn 角色 */
IF CAST(SERVERPROPERTY('IsHadrEnabled') AS INT) = 1
    SELECT
        ag.name                 AS AvailabilityGroup,
        ar.replica_server_name  AS ReplicaName,
        ars.role_desc           AS Role,
        ar.availability_mode_desc AS SyncMode,
        ar.failover_mode_desc     AS FailoverMode
    FROM sys.availability_groups ag
    JOIN sys.availability_replicas ar ON ar.group_id = ag.group_id
    JOIN sys.dm_hadr_availability_replica_states ars ON ars.replica_id = ar.replica_id;
GO

/* 8) CPU / 内存 */
SELECT
    cpu_count                AS LogicalCores,
    hyperthread_ratio,
    physical_memory_kb/1024  AS PhysicalMemoryMB,
    committed_target_kb/1024 AS TargetServerMemoryMB,
    sqlserver_start_time
FROM sys.dm_os_sys_info;

SELECT
    CAST(SUM(awe_allocated_kb+virtual_memory_committed_kb+shared_memory_committed_kb)/1024.0 AS DECIMAL(18,2)) AS TotalServerMemoryMB
FROM sys.dm_os_memory_clerks;
GO

/* 9) Lock Pages In Memory */
SELECT locked_page_allocations_kb/1024 AS LockedPagesMB
FROM sys.dm_os_process_memory;
GO

/* 10) IO Stall（采集时刻快照） */
SELECT
    DB_NAME(vfs.database_id) AS DatabaseName,
    mf.type_desc             AS FileType,
    vfs.num_of_reads,
    vfs.io_stall_read_ms,
    CASE WHEN vfs.num_of_reads = 0 THEN 0
         ELSE vfs.io_stall_read_ms  * 1.0 / vfs.num_of_reads END AS AvgReadStallMs,
    vfs.num_of_writes,
    vfs.io_stall_write_ms,
    CASE WHEN vfs.num_of_writes = 0 THEN 0
         ELSE vfs.io_stall_write_ms * 1.0 / vfs.num_of_writes END AS AvgWriteStallMs
FROM sys.dm_io_virtual_file_stats(NULL, NULL) vfs
JOIN sys.master_files mf
     ON mf.database_id = vfs.database_id AND mf.file_id = vfs.file_id
WHERE DB_NAME(vfs.database_id) LIKE 'MSCRM%' OR vfs.database_id = 2;
GO
```

### 1.2 PowerShell 包装脚本

保存为 `C:\DBA\scripts\Collect-SqlConfig.ps1`，会：

1. 对所有 SQL 实例执行上述 SQL
2. 把结果合并成符合 `memory/sql_config.json` schema 的 JSON
3. 输出到 `D:\DBA\insagent-data\_memory\sql_config.json`

```powershell
param(
    [string[]]$SqlInstances = @("SQL01", "SQL02"),
    [string]$OutFile        = "D:\DBA\insagent-data\_memory\sql_config.json",
    [string]$SqlScriptPath  = "C:\DBA\scripts\Collect-SqlConfig.sql"
)

$ErrorActionPreference = "Stop"
Import-Module SqlServer -ErrorAction SilentlyContinue

$result = @{
    _description   = "SQL Server 实际运行配置快照 - auto generated"
    _last_updated  = (Get-Date -Format "yyyy-MM-ddTHH:mm:ssK")
    _collected_by  = "Collect-SqlConfig.ps1"
    instances      = @()
}

foreach ($srv in $SqlInstances) {
    Write-Host "[*] Collecting $srv ..."
    $ds = Invoke-Sqlcmd -ServerInstance $srv -InputFile $SqlScriptPath -TrustServerCertificate -OutputAs DataTables
    # Invoke-Sqlcmd 返回多个结果集时，$ds 是 DataTable 数组，按上面 SQL 的顺序：
    # 0=version, 1=sp_configure, 2=tempdb, 3=db options, 4=db size, 5=trace flags,
    # 6=alwayson (可能为空), 7=sys_info, 8=total_mem, 9=locked_pages, 10=io_stall
    $ver    = $ds[0]
    $cfg    = $ds[1]
    $tempdb = $ds[2]
    $dbopts = $ds[3]
    $dbsize = $ds[4]
    $tflags = $ds[5]
    $ag     = if ($ds.Count -gt 10) { $ds[6] } else { $null }
    $offset = if ($ag) { 0 } else { -1 }
    $sysinfo= $ds[7 + $offset]
    $totmem = $ds[8 + $offset]
    $locked = $ds[9 + $offset]
    $iost   = $ds[10 + $offset]

    # 转 hashtable 便于查找
    $cfgMap = @{}
    foreach ($r in $cfg) { $cfgMap[$r.name] = [int]$r.value_in_use }

    $inst = [ordered]@{
        server             = $ver[0].ServerName
        instance           = $ver[0].InstanceName
        product_version    = $ver[0].ProductVersion
        product_level      = $ver[0].ProductLevel
        edition            = $ver[0].Edition
        configurations     = @{
            max_server_memory_mb              = $cfgMap["max server memory (MB)"]
            min_server_memory_mb              = $cfgMap["min server memory (MB)"]
            max_degree_of_parallelism         = $cfgMap["max degree of parallelism"]
            cost_threshold_for_parallelism    = $cfgMap["cost threshold for parallelism"]
            optimize_for_ad_hoc_workloads     = $cfgMap["optimize for ad hoc workloads"]
            max_worker_threads                = $cfgMap["max worker threads"]
            backup_compression_default        = $cfgMap["backup compression default"]
            fill_factor_pct                   = $cfgMap["fill factor (%)"]
            remote_query_timeout              = $cfgMap["remote query timeout (s)"]
        }
        tempdb = @{
            files                  = @($tempdb | ForEach-Object {
                @{ name=$_.LogicalName; size_mb=$_.SizeMB; max_size_mb=$_.MaxSizeMB; growth_mb=$_.GrowthMB; physical_path=$_.PhysicalPath }
            })
            total_data_file_count  = ($tempdb | Where-Object FileType -eq 'ROWS' | Measure-Object).Count
            total_data_size_mb     = ($tempdb | Where-Object FileType -eq 'ROWS' | Measure-Object SizeMB -Sum).Sum
            log_size_mb            = ($tempdb | Where-Object FileType -eq 'LOG'  | Measure-Object SizeMB -Sum).Sum
            all_files_equal_size   = (($tempdb | Where-Object FileType -eq 'ROWS' | Select-Object -ExpandProperty SizeMB | Sort-Object -Unique).Count -eq 1)
        }
        databases = @($dbopts | ForEach-Object {
            $sz = $dbsize | Where-Object DatabaseName -eq $_.name | Select-Object -First 1
            @{
                name                           = $_.name
                is_read_committed_snapshot_on  = [bool]$_.is_read_committed_snapshot_on
                snapshot_isolation_state       = "$($_.snapshot_isolation_state_desc)"
                recovery_model                 = "$($_.recovery_model_desc)"
                compatibility_level            = [int]$_.compatibility_level
                auto_shrink                    = [bool]$_.is_auto_shrink_on
                auto_close                     = [bool]$_.is_auto_close_on
                auto_create_statistics         = [bool]$_.is_auto_create_stats_on
                auto_update_statistics         = [bool]$_.is_auto_update_stats_on
                auto_update_statistics_async   = [bool]$_.is_auto_update_stats_async_on
                page_verify                    = "$($_.page_verify_option_desc)"
                size_mb                        = if ($sz) { [int]$sz.DataSizeMB } else { 0 }
                log_size_mb                    = if ($sz) { [int]$sz.LogSizeMB  } else { 0 }
                trustworthy                    = [bool]$_.is_trustworthy_on
                service_broker_enabled         = [bool]$_.is_broker_enabled
            }
        })
        trace_flags_global = @($tflags | Where-Object Global -eq 1 | Select-Object -ExpandProperty TraceFlag)
        alwayson = if ($ag) {
            $me = $ag | Where-Object ReplicaName -like "$srv*" | Select-Object -First 1
            @{
                enabled              = $true
                role                 = "$($me.Role)"
                availability_group   = "$($me.AvailabilityGroup)"
                sync_mode            = "$($me.SyncMode)"
                failover_mode        = "$($me.FailoverMode)"
            }
        } else {
            @{ enabled=$false }
        }
        cpu = @{
            logical_cores = [int]$sysinfo[0].LogicalCores
            numa_nodes    = 1
            affinity_mask_auto = $true
        }
        memory = @{
            total_physical_gb     = [math]::Round($sysinfo[0].PhysicalMemoryMB / 1024, 1)
            total_server_memory_mb= [int]$totmem[0].TotalServerMemoryMB
            target_server_memory_mb = [int]$sysinfo[0].TargetServerMemoryMB
            lock_pages_in_memory  = ([int]$locked[0].LockedPagesMB -gt 0)
        }
        io_stall_sample = @{
            _note                     = "采集时刻累计样本"
            read_stall_ms_per_read    = [math]::Round(($iost | Measure-Object AvgReadStallMs  -Average).Average, 2)
            write_stall_ms_per_write  = [math]::Round(($iost | Measure-Object AvgWriteStallMs -Average).Average, 2)
        }
    }

    $result.instances += $inst
}

$OutDir = Split-Path $OutFile -Parent
if (!(Test-Path $OutDir)) { New-Item -ItemType Directory -Path $OutDir | Out-Null }
$result | ConvertTo-Json -Depth 10 | Out-File -FilePath $OutFile -Encoding UTF8
Write-Host "[+] OK -> $OutFile"
```

## 二、d365_custom.json — 半自动

### 2.1 自动建议脚本

保存为 `C:\DBA\scripts\Suggest-D365Custom.ps1`。读取已有的 `plugin_assemblies.csv` 和 `index_existing.csv`，输出推荐的前缀列表，人工确认后合并到 `d365_custom.json`。

```powershell
param(
    [string]$PluginCsv  = "D:\DBA\insagent-data\plugin_scan\$(Get-Date -f yyyy-MM-01)\plugin_assemblies.csv",
    [string]$IndexCsv   = "D:\DBA\insagent-data\sql_index\index_existing.csv",
    [string]$OutFile    = "D:\DBA\insagent-data\_memory\d365_custom_suggest.json"
)

$result = @{ _note = "人工审核后合并到 memory/d365_custom.json"; suggestions = @{} }

# --- 1) Plugin assembly 前缀推断 ---
if (Test-Path $PluginCsv) {
    $asm = Import-Csv $PluginCsv
    # 按第一个点之前的词聚合
    $prefixes = $asm | ForEach-Object {
        if ($_.Name -match '^([A-Za-z0-9]+\.)') { $matches[1] }
    } | Where-Object { $_ -notmatch '^(Microsoft|System)\.' } |
        Group-Object | Sort-Object Count -Descending |
        Select-Object -First 5 Name, Count
    $result.suggestions.plugin_assembly_prefixes = $prefixes
}

# --- 2) 索引命名前缀 ---
if (Test-Path $IndexCsv) {
    $idx = Import-Csv $IndexCsv
    $idxPrefixes = $idx | ForEach-Object {
        if ($_.IndexName -match '^([A-Za-z_]+_)') { $matches[1] }
    } | Group-Object | Sort-Object Count -Descending |
        Select-Object -First 10 Name, Count
    $result.suggestions.index_name_prefixes = $idxPrefixes
}

# --- 3) 自定义属性前缀（publisher） ---
# 无法纯从采集 CSV 推断，但可以从 index 列名里看 new_ / xxx_ 分布
if (Test-Path $IndexCsv) {
    $colPrefix = $idx.IndexKeyColumns -split '[, ]+' |
                 Where-Object { $_ -match '^([a-z]+)_' } |
                 ForEach-Object { $matches[1] + '_' } |
                 Group-Object | Sort-Object Count -Descending |
                 Select-Object -First 5 Name, Count
    $result.suggestions.custom_attribute_prefixes = $colPrefix
}

$OutDir = Split-Path $OutFile -Parent
if (!(Test-Path $OutDir)) { New-Item -ItemType Directory -Path $OutDir | Out-Null }
$result | ConvertTo-Json -Depth 6 | Out-File $OutFile -Encoding UTF8
Write-Host "[+] 建议已输出: $OutFile  请人工审核后合并到 memory/d365_custom.json"
```

### 2.2 调度

不需要定时。每次 Plugin 扫描（每月 1 号）结束后手动运行一次做建议。

## 三、business_context.json — 完全手动

此文件反映**客户业务语义**（工作时段、SLA、批处理作业），无法从服务器采集。

### 3.1 首次填写清单

请运维/业务协作完成以下信息，填入 `memory/business_context.json`：

1. **业务峰值时段**：工作日 9-11:30、13:30-17:00 是否就是高峰？是否有夜间批处理？
2. **SLA 要求**：响应时间 P95/P99 目标？可用性目标？
3. **维护窗口**：每日备份时段、每周索引重建时段、每月补丁时段
4. **已知 batch job 清单**：月结、ETL、归档作业的时间和影响
5. **关键 URL**：业务方最关心的 endpoint
6. **用户并发**：日常/峰值并发数

### 3.2 建议审查周期

每季度或系统重大变更后，人工 review 一次。

## 四、thresholds.json — 手动但有默认模板

已内置面向中等规模 D365 的默认值。按客户环境调整：

- 小规模（<50 并发）：CPU warning 调低到 60%、slow SQL 阈值调低到 300ms
- 大规模（>200 并发）：CPU warning 调高到 80%、批处理时段容忍度提高
- 敏感业务（金融）：P95 响应目标 1000ms、5xx 比例 warning 0.5%

### 4.1 调整流程

1. 初次巡检用默认值跑 `health_report`
2. 观察告警命中率：
   - 误报多 → 调高 warning 阈值
   - 漏报多 → 调低 warning 阈值
3. 每季度 review 一次
